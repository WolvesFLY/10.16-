#include<stdio.h>
int main() {
	int m,n,sum;
	scanf("%d%d",&m,&n);
	int min=m<n?m:n;
	int max=m>n?m:n;
	for(int i=min; i>=1; i--) {
		if(m%i==0&&n%i==0)
			{sum=i;
		printf("%d ",i);
		break;}
	}
	printf("%d",m*n/sum);

	return 0;
}
以下是对这段代码的解释：
一、整体功能
这段 C 语言代码的作用是求两个整数的最大公约数和最小公倍数，并依次输出最大公约数和最小公倍数。
二、具体步骤
定义变量并读取输入：
int m, n, sum;：定义了三个整数变量 m、n 和 sum，分别用于存储两个输入的整数和最大公约数。
scanf("%d%d", &m, &n);：从用户输入读取两个整数存入变量 m 和 n。
确定较大值和较小值：
int min = m < n? m : n;：使用条件运算符（三目运算符）确定 m 和 n 中的较小值存入变量 min。
int max = m > n? m : n;：同理，确定较大值存入变量 max。
求最大公约数：
通过一个循环从较小值开始递减遍历，寻找能同时整除 m 和 n 的最大整数 i。
for(int i = min; i >= 1; i--)：从较小值开始递减循环。
if(m % i == 0 && n % i == 0)：如果 i 能同时整除 m 和 n，则 i 是它们的公约数。将 i 的值赋给 sum，并输出 i，然后使用 break 语句跳出循环，因为已经找到了最大公约数。
求最小公倍数：
根据最大公约数和两个数的乘积关系，最小公倍数等于两数之积除以最大公约数。
printf("%d", m * n / sum);：输出最小公倍数。
三、示例
例如输入 m = 12，n = 18。
首先确定 min = 12，max = 18。
循环从 12 开始递减，当 i = 6 时，满足 12 % 6 == 0 && 18 % 6 == 0，所以最大公约数 sum = 6，并输出 6。
计算最小公倍数为 12 * 18 / 6 = 36，并输出 36。
